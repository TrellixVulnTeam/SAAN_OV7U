from functools import partial


def mean_aggregation(embeddings):
    embeddings = np.mean(embeddings.reshape(-1, EMBEDDINGS_DIM), axis=0)
    return embeddings

def weighted_aggregation(embeddings):
    embeddings = embeddings.reshape((1,-1,EMBEDDINGS_DIM))
    aggregated = aggregator({'embeddings':embeddings})['aggregated_embeddings']
    return aggregated

def multi_weighted_aggregation(embeddings,
                               biggest_component=False,
                               binarization_threshold = .9):

    embeddings = embeddings.reshape((-1, EMBEDDINGS_DIM))
    
    if len(embeddings) == 1:
        return embeddings.flatten()
    
    try:
   
        aggregated_meta = aggregator({'embeddings':embeddings,
                                     'binarization_threshold':binarization_threshold})
    except Exception as e:
        print(embeddings.shape)
        print(embeddings)
        raise e

    cliques_size = aggregated_meta['clique_size']
    norms = np.linalg.norm(embeddings, axis=1)
    arg_max_norm = np.argsort(norms)[::-1][0]
    
    bmask = aggregated_meta['binary_mask']
    index_to_take =  np.argwhere(bmask[...,arg_max_norm]!=0).flatten()  

    biggest_clique_ind = np.argsort(cliques_size)[::-1][0]
    indecies = biggest_clique_ind if biggest_component else index_to_take
    aggregated = aggregated_meta['aggregated_embeddings'][indecies]
    aggregated = aggregated.reshape((-1,512))
    assert(len(aggregated)==1) 
    return aggregated

def aggregate_template(grouped_df,
                       annotations_df = concated_annotations,
                       embeddings = concated_embeddings,
                       agg_function = mean_aggregation):
    
#     selected_frames = annotations_df.merge(grouped_df, on=['FILENAME','SUBJECT_ID'])
    identity = np.unique(grouped_df['SUBJECT_ID'])
    selected_frames = annotations_df[annotations_df.SUBJECT_ID.isin(identity)]
    selected_frames = selected_frames[selected_frames.FILENAME.isin(grouped_df.FILENAME)]
#     print(len(selected_frames))
    indecies = selected_frames.index.values
#     print(indecies)
    
#     assert(len(identity)<=1)
#     print(indecies)
    if len(indecies):
        embeddings_template = embeddings[indecies]
        aggregated_embedding = agg_function(embeddings_template)
        return aggregated_embedding
    else: return None


def generate_templates(template_df,
                       grouper_key="TEMPLATE_ID",
                       agg_function= weighted_aggregation):
    
    template_df = template_df.groupby(grouper_key)
    aggregated_out = template_df.apply(partial(aggregate_template,
                                               agg_function=agg_function))
    return aggregated_out

def find_ditance(row):
    emb1, emb2 = row.enrolled_agg_emb, row.verif_agg_emb
    emb1 = emb1.reshape(-1, EMBEDDINGS_DIM)
    emb2 = emb2.reshape(-1, EMBEDDINGS_DIM)
    return cosine_distances(emb1, emb2).flatten()

def parallelize(data,
                func,
                num_of_processes=8):
    
    data_split = np.array_split(data, num_of_processes)
    pool = Pool(num_of_processes)
    data = pd.concat(pool.map(func, data_split))
    pool.close()
    pool.join()
    return data

def run_on_subset(func, data_subset):
    return data_subset.apply(func,
                             axis=1)

def parallelize_on_rows(data, func, num_of_processes=8):
    return parallelize(data, partial(run_on_subset, func), num_of_processes)

def auc_from_fpr_tpr(fpr,
                     tpr,
                     trapezoid=False):
    inds = [i for (i, (s, e)) in enumerate(zip(fpr[: -1], fpr[1: ])) if s != e] + [len(fpr) - 1]
    fpr, tpr = fpr[inds], tpr[inds]
    area = 0
    ft = list(zip(fpr, tpr))
    for p0, p1 in zip(ft[: -1], ft[1: ]):
        area += (p1[0] - p0[0]) * ((p1[1] + p0[1]) / 2 if trapezoid else p0[1])
    return area
